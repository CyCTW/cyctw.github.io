vector<int> prefix_function(string s) {
    int n = s.size();
    vector<int> f(n);
    for(int i=0; i<n; i++) {
        for(int k=0; k<=i; k++) {
            if (s.substr(0, k) == s.substr(i - k +1, k)) 
                f[i] = k;
        }
    }
}

vector<int> prefix_function_optimize_2(string s) {
    int n = s.size();

    vector<int> f(n);
    for(int i=1; i<n; i++) {
        int j = f[i-1];
        while(j > 0 && s[i] != s[j]) {
            j = f[j - 1];
        }
        if (s[i] == s[j]) {
            j++;
        }
        f[i] = j;
    }
    return f;
}
